// 顶部跑马灯组件
const NewsTicker = () => {
  const [upcomingEvents, setUpcomingEvents] = useState([]);
  
  useEffect(() => {
    // 获取未来24小时的经济事件
    fetch('/api/economic-calendar/upcoming?hours=24')
      .then(res => res.json())
      .then(data => setUpcomingEvents(data));
      
    // 每分钟更新倒计时
    const interval = setInterval(() => {
      setUpcomingEvents(prev => 
        prev.map(event => ({
          ...event,
          timeRemaining: calculateTimeRemaining(event.releaseTime)
        }))
      );
    }, 60000);
    
    return () => clearInterval(interval);
  }, []);
  
  return (
    <div className="bg-gray-900 text-white py-2 fixed top-0 left-0 right-0 z-50">
      <div className="container mx-auto overflow-hidden">

# 新增API
@app.get("/api/economic-calendar/upcoming")
async def get_upcoming_events(hours: int = 24):
    """获取未来指定小时内的经济事件"""
    # 从日历API获取数据
    events = load_economic_calendar()
    
    # 过滤未来hours小时内的事件
    now = datetime.utcnow()
    cutoff = now + timedelta(hours=hours)
    
    upcoming = [
        e for e in events 
        if now <= datetime.fromisoftime(e["releaseTime"]) <= cutoff
    ]
    
    # 按时间排序
    upcoming.sort(key=lambda x: x["releaseTime"])
    
    # 计算剩余时间
    for e in upcoming:
        release = datetime.fromisoftime(e["releaseTime"])
        e["timeRemaining"] = int((release - now).total_seconds() / 60)
    
    return upcoming

@app.get("/api/topics/heat-analysis")
async def get_heat_analysis():
    """获取双维度热度分析数据"""
    # 加载主题数据
    topics = load_topics()
    
    # 计算当前热度（绝对）
    top_by_heat = sorted(
        topics, 
        key=lambda x: x["heat"], 
        reverse=True
    )[:5]
    
    # 计算热度增长率（环比）
    for topic in topics:
        # 假设有历史热度数据
        history = topic.get("heat_history", [])
        if len(history) >= 2:
            current = history[-1]["heat"]
            previous = history[-2]["heat"]
            if previous > 0:
                topic["growthRate"] = round((current - previous) / previous * 100)
            else:
                topic["growthRate"] = 0
        else:
            topic["growthRate"] = 0
    
    top_by_growth = sorted(
        [t for t in topics if t["growthRate"] > 0],
        key=lambda x: x["growthRate"],
        reverse=True
    )[:5]
    
    return {
        "topByHeat": top_by_heat,
        "topByGrowth": top_by_growth
    }

@app.get("/api/historical/similar/{topic_id}")
async def get_similar_historical(topic_id: str, limit: int = 3):
    """获取与当前主题相似的历史案例"""
    # 加载所有历史案例
    cases = load_historical_cases()
    
    # 基于标签和关键词匹配相似度
    current_topic = get_topic(topic_id)
    
    similar = []
    for case in cases:
        # 计算相似度（简化版）
        similarity = calculate_similarity(current_topic, case)
        if similarity > 0.3:  # 阈值
            similar.append({
                **case,
                "similarity": round(similarity * 100)
            })
    
    # 按相似度排序
    similar.sort(key=lambda x: x["similarity"], reverse=True)
    
    return similar[:limit]
